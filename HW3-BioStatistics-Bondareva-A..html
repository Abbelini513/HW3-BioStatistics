2. diabetes. age age_group, : 2135, 3649 50+. .

#   `diabetes`  character  factor:
pima_data <- pima_data %>%
  mutate(diabetes = as.factor(diabetes))
#      ,    :
diabetes_stats <- pima_data %>%
  group_by(diabetes) %>%
  summarise(
    Absolute_Frequency = n(), #  .
    Relative_Frequency = n() / nrow(pima_data) #  .
  ) %>%
  mutate(
    Percentage = Relative_Frequency * 100 #   =   * 100%.
  )

#     ,    .
print(diabetes_stats)

#              . ,  1%  99%       .
triceps_limits <- quantile(pima_data$triceps, probs = c(0.01, 0.99), na.rm = TRUE)
mass_limits <- quantile(pima_data$mass, probs = c(0.01, 0.99), na.rm = TRUE)

#         .
pima_data_clean <- pima_data %>%
  filter(
    triceps > triceps_limits[1] & triceps < triceps_limits[2], #    `triceps`.
    mass > mass_limits[1] & mass < mass_limits[2], #    mass
    !is.na(triceps), #     `triceps`.
    !is.na(mass) #     `mass`.
  )
